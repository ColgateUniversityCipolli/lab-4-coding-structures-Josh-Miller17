\section{Recipe One}\label{recipe-one}

\normalsize

\includegraphics[width=0.5\linewidth,height=\textheight,keepaspectratio]{images/BreakfastCheesecake.jpg}

\subsection{Instructions}\label{instructions}

\begin{itemize}
\tightlist
\item
  6 Stella D'oro breakfast treats
\item
  1/3 cup sugar
\item
  1/3 cup butter (melted)
\item
  3 5.3 oz vanilla yogurt cups
\item
  1 cup cream cheese (not whipped)
\item
  2 eggs
\item
  2/3 cup sugar
\item
  1/4 cup flour
\item
  1/4 teaspoon salt
\item
  1 heaping teaspoon vanilla bean paste
\item
  Various fresh berries, cut into bite-size pieces
\end{itemize}

\subsection{Instructions}\label{instructions-1}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Place Stella D'oro cookies into a Ziploc bag and pulverize.
\item
  Combine cookie dust, sugar, and melted butter and combine.
\item
  Push the crust into the bottom of an 8 or 9 inch springform pan.
\item
  Preheat oven to 325F.
\item
  Mix the vanilla yogurt cups, cream cheese, eggs, sugar, flour, salt,
  vanilla bean paste together until smooth. A blender works best, but a
  hand mixer can work.
\item
  Pour yogurt mixture on top of the crust in the springform pan.
\item
  Bake for 55 minutes. Cook a few more minutes if the middle is not yet
  set.
\item
  Cool on the counter for about two hours, and then overnight in the
  refrigerator.
\item
  Top with berries and serve.
\end{enumerate}

\section{Recipe Two}\label{recipe-two}

\normalsize

\subsection{Instructions}\label{instructions-2}

\begin{itemize}
\tightlist
\item
  1 bag baby arugula (regular is fine, too)
\item
  1 pound pasta (e.g., bucatini or farfalle)
\item
  6-8 ounces of fresh or jarred pesto
\item
  1 lemon
\item
  Grated parmesan cheese
\item
  Pine nuts
\end{itemize}

\subsection{Instructions}\label{instructions-3}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Boil and cook pasta according to box, strain but reserve 1 cup pasta
  water.
\item
  In the stock pot used to cook the pasta, add two tablespoons of olive
  oil and whisk in pesto over medium heat.
\item
  Whisk in 1/2 cup pasta water, a little more if needed to create a
  light sauce.
\item
  Remove sauce from heat and toss cooked pasta in the sauce.
\item
  Stir arugula in and cover until wilted.
\item
  Add juice from half a lemon, add more to taste.
\item
  Serve with parmesan cheese and pine nuts.
\end{enumerate}

\section{Breakfast Recipe}\label{breakfast-recipe}

\normalsize

\subsection{Instructions}\label{instructions-6}

\begin{itemize}
\tightlist
\item
  1 eggplant
\item
  4 garlic cloves
\item
  4 ounces shredded mozzarella cheese
\item
  1/4 cup panko breadcrumbs
\item
  1 shallot
\item
  2 tsp Italian seasoning
\item
  1 can of fire roasted diced tomatoes
\item
  1 cup couscous
\item
  1 lemon
\item
  1 0.25 oz package of fresh basil
\end{itemize}

\subsection{Instructions}\label{instructions-7}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Cut ends off eggplant, peel, and slice eggplant lengthwise into 4
  slices. Mince garlic and shallot. Zest lemon, cut into wedges adn
  juice half of them. Roughly chop basil.
\item
  Coat eggplant in olive oil and season with salt and pepper. Cook in an
  oven at 425 degrees for 20 minutes, flipping halfway.
\item
  Mix half the garlic, and the mozzarella, breadcrumbs and 1 tablespoon
  olive oil.
\item
  Cook remaining garlic, shallot in olive oil. Add Italian seasoning and
  cook until fragrant.
\item
  Add tomatoes, and a pinch of salt and pepper. Simmer until thickened,
  about 10 minutes.
\item
  Cook couscous according to package instructions. Remove from heat and
  stir in lemon zest and juice, half the basil, and a tablespoon of
  olive oil.
\item
  Top eggplant with tomato sauce and breadcrumbs, and cook an additional
  5-10 minutes.
\item
  Serve topped with remaining basil and lemon.
\end{enumerate}
